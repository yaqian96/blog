```js
//如果i和j指向的字一样，则i不动，j后移
//如果指向的字不一样，说明他们之间的字是连续相同的，让i==j,j后移动
var str='aaabbbbcccccdddd'
var i=0;
var j=1;
let maxCount=0;
let maxChar=''
while(i<=str.length-1){
   if(str[i]!==str[j]){
    console.log（"i和j之间的重复次数j-i次"）
    if (j-i > maxCount){
      maxCount=j-i;
      maxCount=str[i];
     }
      i=j;
   }
   j++;
} 
```
